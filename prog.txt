#include "stdio.h"
main()
{
    int lim,arr[50],del,i,count=0;
    printf("Enter the Limit:");
    scanf("%d",&lim);
    printf("Enter %d Elements then:",lim);

    for(i=0;i<lim;i++)
        scanf("%d",&arr[i]);

    printf("Enter the number to be deleted:");
    scanf("%d",&del);

    for(i=0;i<lim;i++)
    {
        if(arr[i]==del)
        {
            count++;
            while(i<lim-1)
            {
                arr[i]=arr[i+1];
                i++;
            }
        }
    }
    if (count==1)
    {
        for(i=0;i<lim-1;i++)
            printf("%d ",arr[i]);
    }
    else{
        printf("The element you want to delete is not present in the array");
    }
}
